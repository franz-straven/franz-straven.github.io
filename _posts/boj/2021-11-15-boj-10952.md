### 2.2.1 Python

```python
while True:
    A, B = map(int, input().split())
    if (A == 0 and B == 0):
        break
    else:
        print(A + B)
```

### 2.2.2 JavaScript

```jsx
const fs = require('fs');
const input = fs.readFileSync('/dev/stdin').toString().split('\n');

let i = 0;

while(input[i]){
    var arr = input[i].split(' ');
    var a = parseInt(arr[0]);
    var b = parseInt(arr[1]);
    if(a > 0 && b > 0){
        console.log(a + b);
    }
    i++;
}
```
